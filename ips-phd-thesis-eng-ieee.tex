\frontmatter
%
\selectlanguage{american}

%
% Title pages
\input{text-eng/front-title-pages}
%

\input{text-eng/front-dedication}
%

\input{text-eng/front-acknowledgments}

\selectlanguage{american}
%

\maketoc
%

\makelof
%
% List of tables (required if the thesis contains tables)
\makelot
%
% List of algorithms (required if the thesis contains algorithms)
\makeloa
%
% Abbreviations (optional)
\input{text-eng/front-abbreviations}
%
% Symbols (optional)
\input{text-eng/front-symbols}
%
% Glossary (optional)
\input{text-eng/front-glossary}
%--------------------------------------------------------------------------------------------------
%
% MAIN MATTER
%--------------------------------------------------------------------------------------------------
%
\mainmatter

% Introduction
\input{text-eng/main-introduction-phd}

%\part{First Part of the Thesis}
%
% Second chapter
\input{text-eng/main-chapter-2}

%
% Fourth chapter
\input{text-eng/main-chapter-4}
%
% Fifth chapter
\input{text-eng/main-chapter-5}

% Sixth chapter
\input{text-eng/main-chapter-6}

% regard all parts
%\chapteroutsidepart
%
% Conclusions
\input{text-eng/main-conclusions}
%--------------------------------------------------------------------------------------------------
%
% APPENDICES (optional)
%--------------------------------------------------------------------------------------------------
%
\appendix
\begin{appendices}

% For example, proofs of theorems could be an appendix
\input{text-eng/appendix-proofs}

\end{appendices}

%

%--------------------------------------------------------------------------------------------------

\backmatter

\printreferences

\input{text-eng/back-bibliography-ieee}

\input{text-eng/back-biography}
